#include<bits/stdc++.h>
using namespace std;
 
int main()
{
   int n,i,a[100],b[100],c=0,no_of_zeros=0,sum=0,max=INT_MIN,j;
   cin>>n;
   for(i=0;i<n;i++)
   {
   cin>>a[i];
   if(a[i]==0)
   {
    no_of_zeros++;   
   }
   }
   if(no_of_zeros==1)
   cout<<n-1;
   else
   if(no_of_zeros==n)
   cout<<0;
   else
   if(no_of_zeros==n-1)
   cout<<1;
   else
   if(no_of_zeros==0)
   cout<<n;
   else
   {
   for(i=0;i<n;i++)
   {
       if(a[i]==1)
       {
           c++;
       }
       else
       {
           c=0;
       }
       b[i]=c;
   }
   for(i=n-1;i>=0;i--)
   {
       if(b[i]!=0)
       {
           j=i-b[i]-1;
              if(b[j]!=0&&j>=0)
              sum=b[i]+b[j]+1;
              else
              {
                if(b[i]+no_of_zeros==n)
                  sum=b[i];
                  else
                  sum=b[i]+1;
              }
           if(sum>max)
           max=sum;
           i=j+1;
       }
   }
   cout<<max;
   }
    return 0;
}
